## 题目：[84. 柱状图中最大的矩形](https://leetcode-cn.com/problems/largest-rectangle-in-histogram/)

难度困难1474

给定 *n* 个非负整数，用来表示柱状图中各个柱子的高度。每个柱子彼此相邻，且宽度为 1 。

求在该柱状图中，能够勾勒出来的矩形的最大面积。

 

**示例 1:**

![img](pic/histogram.jpg)

```
输入：heights = [2,1,5,6,2,3]
输出：10
解释：最大的矩形为图中红色区域，面积为 10
```

**示例 2：**

![img](pic/histogram-1.jpg)

```
输入： heights = [2,4]
输出： 4
```

 

**提示：**

- `1 <= heights.length <=105`
- `0 <= heights[i] <= 104`





## 解答

```C++
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        // 单调栈
        stack<int> st;

        // 头尾插入0
        heights.insert(heights.begin(), 0);
        heights.push_back(0);

        st.push(0);
        int res = 0;
        for(int i=1; i<heights.size(); ++i) {
            if(heights[i] > heights[st.top()]) {
                st.push(i);
            } else if (heights[i] == heights[st.top()]) {
                st.pop();
                st.push(i);
            } else {
                while (heights[i] < heights[st.top()]) {
                    int mid = st.top();
                    st.pop();
                    int left = st.top();
                    int right = i;
                    int w = right - left - 1;
                    int h = heights[mid];
                    res = max(res, w * h);
                }
                st.push(i);
            }
        }
        return res;    
    }

    int largestRectangleArea2(vector<int>& heights) {
        // 动态规划
        int n = heights.size();
        if(n==0) return 0;
        vector<int> leftMinIndex(n, -1);
        vector<int> rightMinIndex(n, n);

        for(int i=1; i<n; ++i) {
            int t = i - 1;
            while(t >= 0 && heights[t]>=heights[i]) t = leftMinIndex[t];
            leftMinIndex[i] = t;
        }

        for(int i=n-2; i>=0; i--){
            int t = i+1;
            while(t<n && heights[t]>=heights[i]) t = rightMinIndex[t];
            rightMinIndex[i] = t;
        }
        int res = 0;
        for(int i=0; i<n; ++i) {
            int sum = heights[i] * (rightMinIndex[i] - leftMinIndex[i]- 1);
            res = max(res, sum);
        }

        return res;
    }

    int largestRectangleArea1(vector<int>& heights) {
        // 双指针解法
        int maxRes = 0;
        int left, right;
        for(int i=0; i<heights.size(); ++i) {
            left = i;
            right = i;
            for(; left>=0 && heights[left]>=heights[i]; left--) ;
            for(; right<heights.size() && heights[right]>= heights[i]; right++);
            
            maxRes = max(maxRes, (right - left - 1) * heights[i]);
        }       
        return maxRes;
    }
};
```
