## 题目

[128. 最长连续序列](https://leetcode-cn.com/problems/longest-consecutive-sequence/)

难度困难759

给定一个未排序的整数数组 `nums` ，找出数字连续的最长序列（不要求序列元素在原数组中连续）的长度。

 

**进阶：**你可以设计并实现时间复杂度为 `O(n)` 的解决方案吗？

 

**示例 1：**

```
输入：nums = [100,4,200,1,3,2]
输出：4
解释：最长数字连续序列是 [1, 2, 3, 4]。它的长度为 4。
```

**示例 2：**

```
输入：nums = [0,3,7,2,5,8,4,6,0,1]
输出：9
```

 

**提示：**

- `0 <= nums.length <= 104`
- `-109 <= nums[i] <= 109`



## 解答



#### C++

```C++
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {

        // 方法1
        if(nums.empty()) return 0;
        set<int> s(nums.begin(), nums.end());
        int res = 1;
        int count=1;

        for (auto it = s.begin(); it != s.end(); ++it)
        {
            auto tmpIt = it;
            if (++tmpIt != s.end())
            {
                if (*(tmpIt) - *it == 1) {
                    count++;
                    res = count > res ? count : res;
                }
                else {
                    count = 1;
                }
            }
        }
        return res;

        // 方法2
        if(nums.empty()) return 0;
        unordered_map<int, bool> used;

        for(auto i: nums) used[i]=false;

        int longest=0;

        for(auto i : nums){
            if(used[i]) continue;

            int length = 1;

            used[i]=true;

            for(int j=i+1; used.find(j)!=used.end(); ++j){
                used[j]=true;
                ++length;
            }
            for(int j=i-1; used.find(j)!=used.end(); --j){
                used[j]=true;
                ++length;
            }

            longest = max(longest, length);
        }
        return longest;
    }
};
```

//先定义结点 哈希表 以及哈希表相关的基础函数
struct MyNode{
    int data;
    struct MyNode* next;
};
struct HashTable{
    int TableSize;
    struct MyNode** TheLists;
};
// 下一个素数。哈希表的容量
int NextPrime(int n)
{
    if(n%2==0) n++;
    for(;;n+=2)
    {
        for(int i=3;i<=sqrt(n);i++)
           if(n%i==0) goto out;
        return n;
        out:;
    }
}

// 求余数法得到在哈希表中的位置
int Hash(int key, int TableSize){
    int res=abs(key)%TableSize;
    return res;
}

struct HashTable* InitializeHashTable(int size){
    struct HashTable *H = (struct HashTable*)malloc(sizeof(struct HashTable));
    H->TableSize=NextPrime(size);
    H->TheLists = malloc(sizeof(struct MyNode*)*H->TableSize);
    for(int i=0; i<H->TableSize; i++)
    {
        H->TheLists[i] = malloc(sizeof(struct MyNode));
        H->TheLists[i]->next = NULL;
        H->TheLists[i]->data = -1;
    }
    return H;
}

struct MyNode* Find(int key, struct HashTable *H){
    struct MyNode *p, *l;
    l=H->TheLists[Hash(key, H->TableSize)];
    p=l->next;
    while(p!=NULL && p->data!=key) p=p->next;
    return p;
}

void Insert(int key, struct HashTable *H){
    struct MyNode *pos, *l;
    pos=Find(key, H);
    if(pos==NULL){
        struct MyNode *newNode=(struct MyNode *)malloc(sizeof(struct MyNode));
        l=H->TheLists[Hash(key, H->TableSize)];
        newNode->next = l->next;
        l->next = newNode;
        newNode->data = key;
    }
}

void freeHashTable(struct HashTable *H){
    struct MyNode *tempNode, *deleteNode;
    for(int i=0; i<H->TableSize; ++i)
    {
        tempNode=H->TheLists[i];
        while(tempNode!=NULL){
            deleteNode=tempNode;
            tempNode=tempNode->next;
            free(deleteNode);
        }
    }
    free(H);
}

int longestConsecutive(int* a, int n){
    
    
    int res=0;
    struct HashTable *h = InitializeHashTable(n);
    for(int i=0; i<n; i++){
        struct MyNode *p=Find(a[i], h);
        if(p==NULL) Insert(a[i], h);
    }
    for(int i=0; i<n; ++i){
        //只有x-1不存在的时候再往后找(注意：前一项比较条件是确保类型为int的nums[i]在减一后不越界)
        if(a[i]==-2147483647 || Find(a[i]-1, h)==NULL){
            int curMax=0, temp=0;
            while(Find(a[i]+temp, h)!=NULL){
                curMax++;
                temp++;
            }
            res=fmax(curMax, res);
        }
    }
    return res;
}



// 快排
int cmp(const void*a, const void*b)
{
    return (long long)*(int*)a - (long long)*(int*)b;
}

int longestConsecutive(int* a, int n)
{
    // 快排
    if (a == NULL || n == 0) {
        return 0;
    }
    qsort(a, n, sizeof(int), cmp);    
    int t =a[0], cnt = 1, max = 1;
    for (int i = 1; i < n; i++) {
        if (a[i] == t) {
            continue; 
        } else if ((long long)a[i] - t == 1LL) {
            t = a[i]; 
            cnt++;
            if (max < cnt) {
                max = cnt; 
            }
        } else {
            t = a[i]; 
            cnt = 1;
        }            
    }
    return max;
}
